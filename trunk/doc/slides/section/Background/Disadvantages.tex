\subsection*{Disadvantages of standard capturing software}
\begin{frame}
\frametitle{Disadvantages of Standard Packet Capturing Stack}
\begin{itemize}
	\item<2-> Memory allocations:
		\begin{itemize}
			\item<2-> For each new received packet a new \emph{mbuf} 
				is allocated\newline
		\end{itemize}


	\item<3-> Too many packet copies:
		\begin{itemize}
\ifthenelse{\boolean{SUMMIT}}{
			\item<3-> DMA: $Controller \Rightarrow RAM$
			\item<3-> BPF: $RAM \Rightarrow RAM (BPF\ Buffer)$
			\item<3-> read(2): $RAM \Rightarrow RAM (Userspace\ Buffer)$\emph{(*)}\newline
}{
			\item<3-> DMA: $FIFO \Rightarrow Packet\ Buffer$
			\item<3-> $Packet\ Buffer \Rightarrow BPF\ Buffer$
			\item<3-> $BPF\ Buffer \Rightarrow Userspace\ Buffer$\emph{(*)}\newline
}
		\end{itemize}

	\item<4-> System calls:
		\begin{itemize}
			\item<4-> User-space application receives the packets using read(2)\emph{(*)}
			\item<4-> Saving packets to the hard disk\newline
		\end{itemize}

\end{itemize}
\begin{tiny}
\emph{(*)} Using Zero-Copy BPF eliminates the last copy and system call
\end{tiny}
\end{frame}
